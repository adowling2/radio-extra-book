\chapter{Solutions to Selected Worked Examples}
\section{Capacitive Reactance}
For \(C=\SI{100}{\pico\farad}\) and \(f=\SI{14.2}{\mega\hertz}\),
\[
X_C=\frac{1}{2\pi f C}=\SI{112}{\ohm}.
\]
The signed impedance is capacitive:
\[
Z_C=-\jj\SI{112}{\ohm}.
\]

\section{RLC Bandwidth}
For \(L=\SI{2.2}{\micro\henry}\), \(C=\SI{150}{\pico\farad}\), and
\(R=\SI{3}{\ohm}\),
\[
f_0\approx \SI{8.76}{\mega\hertz},
\qquad
Q_s\approx 40.4,
\qquad
BW\approx \SI{217}{\kilo\hertz}.
\]

\section{SWR from Reflection Coefficient}
For \(|\Gamma|=0.2\),
\[
\mathrm{SWR}=\frac{1+0.2}{1-0.2}=1.5.
\]
The reflected power fraction is \(|\Gamma|^2=0.04\), or \(4\%\).
